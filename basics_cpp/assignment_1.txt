Q1: Pseudo code to calculate the AREA of a rectangle.

FUNCTION CalculateAreaOfRectangle
     INPUT length, width;

     CHECK if length <= 0 && width <= 0:
               return 0;
     VARIABLE area = length * width;

     return area;
END FUNCTION

Q2: Pseudo code to calculate the AVERAGE of 3 Numbers.

FUNCTION CalculateAverageOfNumbers
     INPUT a, b, c;

     VARIABLE average = (a+b+c) / 3;

     return average;
END FUNCTION


Q3: Pseudo code to calculate the LARGEST of 3 Numbers.

FUNCTION GetLargestNumber
     INPUT a, b, c;

     # method 1
     VARIABLE maximum = max(a, b, c);

     return maximum;

     # method 2

     VARIABLE largest;
     CHECK if a >= b AND a >= c:
               largest = a;

          elseif b >= a AND b >= c:
               largest = b;
          else:
               largest = c;

     return largest;
END FUNCTION

Q4: Pseudo code to print if a number N is EVEN or ODD.

FUNCTION CheckEvenOrOdd
     INPUT number;

     CHECK if number <= 0:
               return;
     CHECK if number % 2 == 0;
               return 'Even';
           else:
               return 'Odd';
END FUNCTION
